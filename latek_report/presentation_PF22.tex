\begin{frame}{Contexte et probl\'{e}matique}
  \begin{columns}[T]
    \begin{column}{0.56\textwidth}
      \textbf{\color{darkblue}Le co\^{u}t de la donn\'{e}e : verrou \'{e}conomique principal}

      \smallskip
      L'entra\^{i}nement de r\'{e}seaux profonds performants n\'{e}cessite de vastes jeux de donn\'{e}es \textbf{annot\'{e}s de haute qualit\'{e}}. Or~:

      \medskip
      \begin{itemize}\setlength\itemsep{5pt}
        \item Photographe professionnel, \'{e}quipement haute r\'{e}solution $\Rightarrow$ co\^{u}t \'{e}lev\'{e}
        \item Annotation manuelle par experts $\Rightarrow$ temps long
        \item Passage \`{a} l'\'{e}chelle industriel difficile
      \end{itemize}

      \begin{alertblock}{Question centrale}
        Peut-on remplacer la majorit\'{e} des images co\^{u}teuses (\textbf{HF}) par des images d\'{e}grad\'{e}es bon march\'{e} (\textbf{BF}) sans sacrifier les performances~?
      \end{alertblock}
    \end{column}
    \begin{column}{0.40\textwidth}
      \begin{tikzpicture}[font=\small]
        \node[rounded corners=4pt, fill=hfcolor!15, draw=hfcolor,
              text width=4.2cm, align=center, minimum height=1.1cm] at (0, 2.8)

        \node[rounded corners=4pt, fill=bfcolor!15, draw=bfcolor,
              text width=4.2cm, align=center, minimum height=1.1cm] at (0, 0.7)

\begin{frame}{Architecture : ResNet-18 \textit{from scratch}}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \textbf{\color{darkblue}Pourquoi ResNet-18 ?}
      \begin{itemize}\setlength\itemsep{4pt}
        \item $\approx$11M param\`{e}tres : compromis expressivit\'{e} / overfitting
        \item \textbf{Connexions r\'{e}siduelles} ($F(x)+x$) : r\'{e}sistance au bruit
        \item Standard acad\'{e}mique reproductible
        \item Poids \textbf{al\'{e}atoires} (\texttt{weights=None}) : mesure l'impact pur du dataset
        \item T\^{e}te adapt\'{e}e : \textbf{10 classes Animals-10}
      \end{itemize}

      \medskip
      \begin{exampleblock}{Hyperparam\`{e}tres}
        \footnotesize
        Loss: \textbf{CrossEntropyLoss} \quad Opt: \textbf{Adam} $lr=0.001$\\
        Batch: \textbf{64} \quad \'{E}poques: \textbf{20} \quad AMP: FP16
      \end{exampleblock}
    \end{column}

\begin{frame}{Perspectives et prochaines \'{e}tapes}
  \begin{columns}[T]
    \begin{column}{0.47\textwidth}
      {\footnotesize\textbf{\color{darkblue}Am\'{e}liorations Strat\'{e}gie 1 (priorit\'{e} HF)}}
      \begin{itemize}\footnotesize\setlength\itemsep{2pt}
        \item Early stopping + checkpointing val HF
        \item Grille $(E_{BF},\,E_{HF})$ pour optimiser dur\'{e}e relative
        \item Gel partiel du backbone en fine-tuning
        \item Augmentations cibl\'{e}es (RandAugment, Mixup)
      \end{itemize}

      \medskip
      {\footnotesize\textbf{\color{darkblue}Am\'{e}liorations Strat\'{e}gie 3 (Noisy Student)}}
      \begin{itemize}\footnotesize\setlength\itemsep{2pt}
        \item Renforcer le Teacher (plus d'\'{e}poques, meilleur LR)
        \item Seuil de confiance pseudo-labels ($>0.7$)
        \item R\'{e}duire $\alpha$ BF (0.2--0.6), augmentations moins agressives
        \item Warm-up Student sur HF seul avant injection BF
      \end{itemize}
    \end{column}
    \begin{column}{0.47\textwidth}
      {\footnotesize\textbf{\color{darkblue}Axes de recherche restants}}
      \begin{itemize}\footnotesize\setlength\itemsep{2pt}
        \item Fronti\`{e}re de Pareto co\^{u}t/performance HF
        \item Variation du ratio HF/BF (5\%, 15\%, 20\%\ldots)
        \item Combinaison Strat\'{e}gie 1 + curriculum pond\'{e}r\'{e}
        \item Architecture backbone alternative (ResNet-34)
      \end{itemize}

      \medskip
      \begin{block}{\footnotesize Objectif de fin de projet}
        \footnotesize D\'{e}montrer rigoureusement qu'une approche multi-fid\'{e}lit\'{e} permet d'atteindre ou d\'{e}passer les performances d'un mod\`{e}le 100\% HF \`{a} \textbf{co\^{u}t d'annotation r\'{e}duit d'au moins 50\%}.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}
